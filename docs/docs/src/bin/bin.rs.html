fn bench_dyn() {
    let start = PreciseTime::now();

    let mut img = image::open("examples/input_images/beach.jpg").unwrap();
    let background_color = Rgb { r: 30, g: 45, b: 10};

    draw_filled_rect_mut(&mut img, 
                        Rect::at(20, 30).of_size(100, 100), 
                        Rgba([background_color.r, background_color.g, 
                        background_color.b, 255u8]));
    let red3 = Rgb {r: 244, g: 80, b: 50};

    let triangle: Triangle = Triangle::new(red3, 50, 80, 100, 150, 200, 180);

    let point = Point::new(triangle.x1, triangle.y1);
    let point2 = Point::new(triangle.x2, triangle.y2);
    let point3 = Point::new(triangle.x3, triangle.y3);

    let points = vec![point, point2, point3];
    let tri1 = Rgb { r: 45, g: 67, b: 129};

    draw_convex_polygon_mut(&mut img, 
                        points.as_slice(), 
                        Rgba([tri1.r, tri1.g, tri1.b, 255u8]));

    let tri = Rgb { r: 123, g: 45, b: 160 };
    draw_convex_polygon_mut(&mut img, 
                        points.as_slice(), 
                        Rgba([tri.r, tri.g, tri.b, 255u8]));

    let pixel =  image::Rgba([background_color.r, background_color.g, background_color.b, 255]);
    let image_buffer = ImageBuffer::from_pixel(img.width() + 50, img.height() + 50, pixel);
    let mut rgba_img = image::ImageRgba8(image_buffer);

    image::imageops::overlay(&mut rgba_img, &img, 100, 200);

    rgba_img.save("DynamicImage.jpg").unwrap();
    let end = PreciseTime::now();
    println!("Took {} seconds to create image.", start.to(end));
}
